$(document).ready(function(e) {

 //$("#header_top_div").hide();

 $(window).scroll(function(){

   var w_pos = $(document).scrollTop();

   if( w_pos > 20){

	    $(".w_pos").html(w_pos); 	 

        $("#header_top_div").show().addClass("div_fixed");

	 }

   else{

	  $("#header_top_div").removeClass("div_fixed"); 

	 }	 

 });   

});

</script>
