<script>

 $(document).ready(function(e) {

   $(document).on('click','#btnUpload',function(){

     var fd = new FormData();

	 var files = $('#fileImage')[0].files[0];

	 var email = $("#email").val();

	 fd.append('email',email);

	 fd.append('file',files);

	 $.ajax({

            url: 'upload.php',

            type: 'post',

            data: fd,

            contentType: false,

            processData: false,

            success: function(response){

                $("#result").html(response);

            }

        });



   });

 });

</script>
