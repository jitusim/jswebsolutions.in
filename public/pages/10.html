<script>
$(document).ready(function(){
    $('#searchBox').on('keyup',function(){
        var searchContent = $(this).val().toLowerCase();
        $('#tableData tbody tr').each(function(){
            var lineStr = $(this).text().toLowerCase();
            if(lineStr.indexOf(searchContent) === -1){
                $(this).hide();
            }else{
                $(this).show();
            }
        });
    });
});
</script>
