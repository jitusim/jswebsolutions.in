<script>
 function pancardValidate(panVal){
	document.getElementById('response').innerHTML = " ";
	var regpan = /^([a-zA-Z]){5}([0-9]){4}([a-zA-Z]){1}?$/;
    if(panVal != " "){ 
	 if(regpan.test(panVal)){
        document.getElementById('response').innerHTML = "<span style='color:green;'>Valid Pan Card Number.</span>";  
	   } else {
	    document.getElementById('response').innerHTML = "<span style='color:red;'>Invalid Pan Card Number .</span>";  
      }
	}
    else{
	 document.getElementById('response').innerHTML = "<span style='color:red;'>Pan Card value is required .</span>";   
	}	
   }
</script>
